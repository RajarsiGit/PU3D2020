CREATE TABLE form (
    id int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    reg_id varchar(12),
    name varchar(100),
    qualification varchar(30),
	designation varchar(255),
    address varchar(255),
    phone varchar(15),
    email varchar(50),
    intend varchar(255),
    accom varchar(255),
    calc varchar(10),
    order_id varchar(20)
);
